package com.greedring.practice01;

import java.text.DateFormat;
import java.text.SimpleDateFormat;
import java.util.Date;

public class MainApp {
    public static void main(String[] args) {
        Date now = new Date();
        String rule = "yyyy年MM月dd日 HH:mm:ss";
        DateFormat df = new SimpleDateFormat(rule);
        System.out.println(df.format(now));
    }
}

训练案例10

  • 训练考核知识点
    Date类、SimpleDateFormat类、异常处理

  • 训练描述
    请用代码实现如下需求:
    把你的生日字符串表示转换为对应的日期对象.

  • 操作步骤描述

1.在com.itheima.level1包下创建Test10类
2.在Test10类中,提供main方法,在main方法中
a)创建SimpleDateFormat对象 df,并制定日期格式
b)调用df的parse(String str)方法,传入你的生日字符串;结束返回的日期对象
c)打印这个日期对象
d)使用try…catch处理解析异常

package com.greedring.practice02;

import java.text.DateFormat;
import java.text.ParseException;
import java.text.SimpleDateFormat;
import java.util.Date;

public class MainApp {
    public static void main(String[] args) throws ParseException {
        Date date = new Date();
        String rule = "yyyy年MM月dd日 HH:mm:ss";
        DateFormat df = new SimpleDateFormat(rule);
        String birthday = "1999年1月15日 00:00:00";
        try {
            date = df.parse(birthday);
        } catch (ParseException e) {
            e.printStackTrace();
        }
        System.out.println(date);
    }
}
