日历类

package com.greedring.test;

import java.util.Calendar;
import java.util.Date;

public class Test {
    public static void main(String[] args) {
        //创建Calendar
        Calendar calendar = Calendar.getInstance();
        //根据字符串获取信息
        System.out.println(calendar.get(Calendar.YEAR));
        System.out.println(calendar.get(Calendar.MONTH));//0-11
        System.out.println(calendar.get(Calendar.DAY_OF_MONTH));
        System.out.println(calendar.get(Calendar.HOUR));
        System.out.println(calendar.get(Calendar.MINUTE));
        System.out.println(calendar.get(Calendar.SECOND));
        System.out.println(calendar.get(Calendar.DAY_OF_WEEK));//5 星期四 1星期日
        //根据Calendar对象获取Date对象
        Date date = calendar.getTime();
        System.out.println(date);
        //set(int field，int value) 设置日历类指定字段
        calendar.set(Calendar.YEAR, 2020);
        Date date1 = calendar.getTime();
        System.out.println(date1);
        //设置年月日
        calendar.set(2020, 10 - 1, 1);
        System.out.println(calendar.getTime());
        //获取当前calendar对象的时间戳
        long timeInMillis = calendar.getTimeInMillis();
        System.out.println(timeInMillis);
        //通过date对象设置指定时间的calendar
        calendar.setTime(new Date());
        System.out.println(calendar.getTime());
        //通过毫秒设置日历
        calendar.setTimeInMillis(160151936106L);
        System.out.println(calendar.getTime());
    }
}

        System.out.println(Math.abs(-4));

        System.out.println(Math.round(4.2));

正则表达式

匹配字符串的一种模式
正则表达式也是一个字符串
//匹配指定单个字符
String regex = “a”;
System.out.println(“a”.matches(regex));//true

//2.匹配字符集、字符组：[abc]
String regex = “abc”;
System.out.println(“a”.matches(regex));
System.out.println(“b”.matches(regex));
System.out.println(“c”.matches(regex));
System.out.println(“abc”.matches(regex));

//3.字符类：[^abc]//除abc以外
regex = “[^abc]”;
System.out.println(“a”.matches(regex));//false
System.out.println(“b”.matches(regex));//false
System.out.println(“c”.matches(regex));//false
System.out.println(“d”.matches(regex));//true
System.out.println(“dd”.matches(regex));//false 只能匹配单个字符

//4.字符类：[a-zA-Z]//表示范围
regex = “[a-z_A-Z0-9]”;
System.out.println(“a”.matches(regex));
System.out.println(“A”.matches(regex));
System.out.println(“1”.matches(regex));
System.out.println(“_”.matches(regex));

//预定义字符
//1.任意字符
regex = “.”;
System.out.println(“x”.matches(regex));//true
//2.数字字符[0-9]
regex = “\d”;
System.out.println(“1”.matches(regex));//true
//3.预定义字符类：\w [a-zA-Z_0-9]
regex = “\w”;
System.out.println(“_”.matches(regex));//true

//4.边界匹配器：\b
regex = “”

//数量词
//1. ? 0个或1个
regex = “a?”;
System.out.println(“a”.matches(regex));//true
System.out.println(“”.matches(regex));//true
//2. * 0个或多个
regex = “a*”;
System.out.println(“aaaa”.matches(regex));//true
System.out.println(“”.matches(regex));//true
//3. + 1个或多个
regex = “a+”;
System.out.println(“aaaa”.matches(regex));//true
System.out.println(“”.matches(regex));//false
//4. 指定具体数量
regex = “a{5}”;//只能五次
//regex = “a{5,}”;至少五次
//regex = “a{5,10}”;五到十次
System.out.println(“aaaaa”.matches(regex));//true
System.out.println(“aaa”.matches(regex));//false
